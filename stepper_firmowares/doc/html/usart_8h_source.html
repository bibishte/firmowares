27 #ifndef USART_H_
28 #define USART_H_
29 /**
30  * @defgroup pfleury_uart UART Library
31  * @code #include <usart.h> @endcode
32  *
33  * @brief Interrupt UART library using the built-in UART with transmit and receive circular buffers.
34  *
35  * This library can be used to transmit and receive data through the built in UART.
36  *
37  * An interrupt is generated when the UART has finished transmitting or
38  * receiving a byte. The interrupt handling routines use circular buffers
39  * for buffering received and transmitted data.
40  *
41  * The UART_RX_BUFFER_SIZE and UART_TX_BUFFER_SIZE constants define
42  * the size of the circular buffers in bytes. Note that these constants must be a power of 2.
43  * You may need to adapt this constants to your target and your application by adding
44  * CDEFS += -DUART_RX_BUFFER_SIZE=nn -DUART_RX_BUFFER_SIZE=nn to your Makefile.
45  *
46  * @note Based on Atmel Application Note AVR306
47  * @author Peter Fleury pfleury@gmx.ch http://jump.to/fleury
48  */

110 /**
111  * @brief Get received byte from ringbuffer
112  *
113  * Returns in the lower byte the received character and in the
114  * higher byte the last receive error.
115  * UART_NO_DATA is returned when no data is available.
116  *
117  * @param void
118  * @return lower byte: received byte from ringbuffer
119  * @return higher byte: last receive status
120  * - \b 0 successfully received data from UART
121  * - \b UART_NO_DATA
122  * <br>no receive data available
123  * - \b UART_BUFFER_OVERFLOW
124  * <br>Receive ringbuffer overflow.
125  * We are not reading the receive buffer fast enough,
126  * one or more received character have been dropped
127  * - \b UART_OVERRUN_ERROR
128  * <br>Overrun condition by UART.
129  * A character already present in the UART UDR register was
130  * not read by the interrupt handler before the next character arrived,
131  * one or more received characters have been dropped.
132  * - \b UART_FRAME_ERROR
133  * <br>Framing Error by UART
134  */
135 int usart_getchar(FILE *stream);
