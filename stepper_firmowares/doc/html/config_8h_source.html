43 #define set_bit(byte, bit) byte |= (1 << bit)
44 #define clr_bit(byte, bit) byte &= ~(1 << bit)

46 /** \def set_as_output
47  * Write logical one bit \c DDxn in IO register \c DDRx.
48  * This will configure Pxn as an output pin. */
49 #define set_as_output(DDRx, DDxn) DDRx |= (1 << DDxn)
50 
51 /** \def set_as_input
52  * Write logical zero bit \c DDxn in IO register \c DDRx.
53  * This will configure Pxn as an input pin. */
54 #define set_as_input(DDRx, DDxn) DDRx &= ~(1 << DDxn)
55 
56 /** \def set_port
57  * Write logic one bit \c PORTxn in IO register \c PORTx.
58  * When the pin is configured as an output pin,
59  * the port pin is driven high (one).*/
60 #define set_port(PORTx, PORTxn) PORTx |= (1 << PORTxn)
61 
62 /** \def clr_port
63  * Write logic zero bit \c PORTxn in IO register \c PORTx.
64  * When the pin is configured as an output pin,
65  * the port pin is driven low (zero).*/
66 #define clr_port(PORTx, PORTxn) PORTx &= ~(1 << PORTxn)
67 
68 /** \def check_pin
69  * Test whether bit \c PINxn in IO register \c PINx is set.
70  * This will return a 0 if the port pin is driven low, and non-zero
71  * if the pin is driven high. */
72 #define check_pin(PINx, PINxn) (PINx & (1 << PINxn))
73 
74 #define CHECKPIN(x,y) ((x&(1<<y))!=0?1:0)
